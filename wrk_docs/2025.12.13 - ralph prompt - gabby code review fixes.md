#### M9: G.711 Encode Non-Standard (call.rs:321-353)

**Solution:** Verify and fix the encoding algorithm.

```rust
// The standard mu-law encoding algorithm:
fn encode_ulaw(sample: i16) -> u8 {
    const BIAS: i16 = 0x84;
    const CLIP: i16 = 32635;

    let sign = if sample < 0 { 0x80u8 } else { 0u8 };
    let mut sample = if sample < 0 { -sample } else { sample };

    if sample > CLIP {
        sample = CLIP;
    }
    sample = sample.saturating_add(BIAS);

    let exponent = (sample as u16).leading_zeros() as u8;
    let exponent = 7u8.saturating_sub(exponent.saturating_sub(16));

    let mantissa = ((sample >> (exponent + 3)) & 0x0F) as u8;

    !(sign | (exponent << 4) | mantissa)
}
```

#### M11: Sentence Boundary Abbreviations (audio.rs:152-169)

**Solution:** Improve the sentence boundary detection.

```rust
pub fn find_sentence_boundary(text: &str) -> Option<usize> {
    // Common abbreviations that don't end sentences
    const ABBREVIATIONS: &[&str] = &[
        "Mr.", "Mrs.", "Ms.", "Dr.", "Prof.", "Sr.", "Jr.",
        "vs.", "etc.", "i.e.", "e.g.", "a.m.", "p.m.",
        "U.S.", "U.K.", "Mt.", "St.", "Ave.", "Blvd.",
    ];

    let chars: Vec<(usize, char)> = text.char_indices().collect();

    for (i, (byte_idx, c)) in chars.iter().enumerate() {
        if *c == '.' || *c == '!' || *c == '?' {
            // Check if this is an abbreviation
            if *c == '.' {
                let before = &text[..byte_idx + 1];
                if ABBREVIATIONS.iter().any(|abbr| before.ends_with(abbr)) {
                    continue;
                }
            }

            // End of string or followed by whitespace
            if i == chars.len() - 1 {
                return Some(byte_idx + c.len_utf8());
            }
            if i + 1 < chars.len() && chars[i + 1].1.is_whitespace() {
                return Some(byte_idx + c.len_utf8());
            }
        }
    }

    None
}
```

fn extract_sdp_media(msg: &str) -> Option<SdpMedia> {
    let sdp_start = msg.find("\r\n\r\n")? + 4;
    let sdp = &msg[sdp_start..];

    let mut connection_ip = None;
    let mut audio_port = None;
    let mut codecs = Vec::new();

    for line in sdp.lines() {
        if line.starts_with("c=IN IP4 ") {
            connection_ip = Some(line[9..].split_whitespace().next()?.to_string());
        } else if line.starts_with("m=audio ") {
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() >= 4 {
                audio_port = parts[1].parse().ok();
                codecs = parts[3..].iter().filter_map(|s| s.parse().ok()).collect();
            }
        }
    }

    Some(SdpMedia {
        ip: connection_ip?,
        port: audio_port?,
        codecs,
    })
}
